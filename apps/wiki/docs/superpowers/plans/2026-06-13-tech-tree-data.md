# Tech-tree data layer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Persist the in-game Trampler research tree (nodes, factions, tiers, unlock costs, prerequisite edges, unlocked items) as structured data so a later feature can render the tree and compute total cost to reach any node.

**Architecture:** A tech node is `Entity(kind:"tech-node")` plus a new `TechNodeStats` sub-table (mirroring `ItemStats`/`TramplerStats`). Relationships use `EntityLink` with three new roles: `tech-unlocks` (node→item/part), `tech-prereq` (node→node edge), `tech-research-cost` (node→item, with `amount`). Data is authored into `prisma/tech-tree.json` (OCR'd from screenshots, user-verified) and ingested by `prisma/seed.ts`, following the existing `tramplers.json`→seed pattern. Pure transforms/validation live in `prisma/seed-transform.ts` and are unit-tested without a DB.

**Tech Stack:** Next.js, Prisma 6 (Postgres/Neon), TypeScript, vitest, tsx.

// --- Tech tree (prisma/tech-tree.json → tech-node entities + EntityLink rows) ---

export interface RawTechCostItem { slug?: string; name: string; amount: number }

export interface RawTechNode {
  slug: string;
  name: string;
  faction: string; // "godlewski" | "kaiser" | "landwehr"
  tier: number; // 1-4
  category: string;
  researchCost?: number;
  sortOrder?: number;
  unlocks?: string[]; // resolved item/part slugs the node grants
  unlocksRaw?: string[]; // verbatim OCR names — review aid only, never seeded
  prereqs?: string[]; // prerequisite tech-node slugs
  researchCostItems?: RawTechCostItem[];
}

export interface TechLinkRow { targetSlug: string | null; name: string; amount: number | null; sortOrder: number }

/** Deterministic node slug: tech-<faction>-t<tier>-<kebab name>. */
export function techNodeSlug(faction: string, tier: number, name: string): string {
  const kebab = name.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-+|-+$/g, "");
  return `tech-${faction}-t${tier}-${kebab}`;
}

/** tech-unlocks rows (node → granted entity), ordered by array position. */
export function unlockRows(node: RawTechNode): TechLinkRow[] {
  return (node.unlocks ?? []).map((slug, i) => ({ targetSlug: slug, name: slug, amount: null, sortOrder: i }));
}

/** tech-prereq rows (node → prerequisite node), ordered by array position. */
export function prereqRows(node: RawTechNode): TechLinkRow[] {
  return (node.prereqs ?? []).map((slug, i) => ({ targetSlug: slug, name: slug, amount: null, sortOrder: i }));
}

/** tech-research-cost rows (node → item required to research), ordered by array position. */
export function researchCostRows(node: RawTechNode): TechLinkRow[] {
  return (node.researchCostItems ?? []).map((c, i) => ({ targetSlug: c.slug ?? null, name: c.name, amount: c.amount, sortOrder: i }));
}

export interface TechTreeIssue { node: string; kind: "error" | "warning"; message: string }

/** Structural validation against the set of known entity slugs (items + trampler-parts).
 *  Errors block ingest; warnings are surfaced for review. */
export function validateTechTree(nodes: RawTechNode[], knownEntitySlugs: Set<string>): TechTreeIssue[] {
  const issues: TechTreeIssue[] = [];
  const nodeSlugs = new Set(nodes.map((n) => n.slug));
  const seen = new Set<string>();
  for (const n of nodes) {
    if (seen.has(n.slug)) issues.push({ node: n.slug, kind: "error", message: "duplicate node slug" });
    seen.add(n.slug);
    if (!["godlewski", "kaiser", "landwehr"].includes(n.faction)) {
      issues.push({ node: n.slug, kind: "error", message: `invalid faction "${n.faction}"` });
    }
    if (!Number.isInteger(n.tier) || n.tier < 1 || n.tier > 4) {
      issues.push({ node: n.slug, kind: "error", message: `tier out of range: ${n.tier}` });
    }
    for (const p of n.prereqs ?? []) {
      if (!nodeSlugs.has(p)) issues.push({ node: n.slug, kind: "error", message: `prereq "${p}" is not a known node` });
    }
    for (const u of n.unlocks ?? []) {
      if (!knownEntitySlugs.has(u)) issues.push({ node: n.slug, kind: "warning", message: `unlock "${u}" has no matching item/part entity` });
    }
    for (const c of n.researchCostItems ?? []) {
      if (c.slug && !knownEntitySlugs.has(c.slug)) issues.push({ node: n.slug, kind: "warning", message: `research-cost item "${c.slug}" has no matching entity` });
    }
  }
  return issues;
}
```

  // --- Tech tree: tech-node entities + TechNodeStats + research links ---
  const techNodes: Record<string, RawTechNode> = JSON.parse(
    readFileSync(join(__dirname, "tech-tree.json"), "utf-8"),
  );
  const nodeList = Object.values(techNodes);

  // Validate against known item/part slugs before writing anything.
  const knownEntitySlugs = new Set(
    (await prisma.entity.findMany({ where: { kind: { in: ["item", "trampler-part"] } }, select: { slug: true } })).map((e) => e.slug),
  );
  const techIssues = validateTechTree(nodeList, knownEntitySlugs);
  for (const i of techIssues) console[i.kind === "error" ? "error" : "warn"](`[tech-tree] ${i.node}: ${i.message}`);
  if (techIssues.some((i) => i.kind === "error")) throw new Error("Tech-tree validation failed (see errors above)");

  const techSlugs: string[] = [];
  for (const n of nodeList) {
    if (!isFaction(n.faction)) throw new Error(`Unknown faction "${n.faction}" for ${n.slug}`);
    if (!isTramplerCategory(n.category) && !isItemCategory(n.category)) {
      throw new Error(`Unknown category "${n.category}" for tech node ${n.slug}`);
    }
    techSlugs.push(n.slug);
    const identity = { name: n.name, category: n.category };
    const stats = { faction: n.faction, tier: n.tier, researchCost: opt(n.researchCost), sortOrder: opt(n.sortOrder) };
    await prisma.entity.upsert({
      where: { slug: n.slug },
      create: { slug: n.slug, kind: "tech-node", ...identity, techNodeStats: { create: stats } },
      update: { ...identity, techNodeStats: { upsert: { create: stats, update: stats } } },
    });
  }
  const prunedTech = await prisma.entity.deleteMany({ where: { kind: "tech-node", slug: { notIn: techSlugs } } });
  if (prunedTech.count > 0) console.log(`Pruned ${prunedTech.count} tech node(s) no longer in the snapshot`);

  // Combined slug → id across items, parts, and nodes (link targets span all three).
  const techIdBySlug = new Map(
    (await prisma.entity.findMany({
      where: { kind: { in: ["item", "trampler-part", "tech-node"] } },
      select: { slug: true, id: true },
    })).map((e) => [e.slug, e.id]),
  );

  // Tech links are snapshot-owned → delete + recreate per role.
  const TECH_ROLES = ["tech-unlocks", "tech-prereq", "tech-research-cost"] as const;
  for (const n of nodeList) {
    const sourceId = techIdBySlug.get(n.slug)!;
    for (const role of TECH_ROLES) {
      await prisma.entityLink.deleteMany({ where: { sourceId, role } });
    }
    const create = async (role: string, rows: TechLinkRow[]) => {
      if (!rows.length) return;
      await prisma.entityLink.createMany({
        data: rows.map((r) => {
          const targetId = r.targetSlug ? techIdBySlug.get(r.targetSlug) ?? null : null;
          if (r.targetSlug && !targetId) console.warn(`[tech-tree] ${role} target "${r.targetSlug}" on ${n.slug} does not resolve`);
          return { sourceId, role, targetId, name: r.name, amount: r.amount, sortOrder: r.sortOrder };
        }),
      });
    };
    await create("tech-unlocks", unlockRows(n));
    await create("tech-prereq", prereqRows(n));
    await create("tech-research-cost", researchCostRows(n));
  }

  const techCount = await prisma.entity.count({ where: { kind: "tech-node" } });
  if (techCount !== nodeList.length) throw new Error(`Tech-node count mismatch: DB ${techCount}, snapshot ${nodeList.length}`);
  console.log(`Seeded ${nodeList.length} tech nodes.`);
```
